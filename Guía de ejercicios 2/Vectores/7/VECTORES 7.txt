int pines[] = {2, 3, 4, 5, 6};

void setup() {
  for (int i = 0; i < sizeof(pines) / sizeof(pines[0]); i++) {
    pinMode(pines[i], OUTPUT);
  }
}

void loop() {
  for (int i = 0; i < sizeof(pines) / sizeof(pines[0]); i++) {
    digitalWrite(pines[i], HIGH);
    delay(200); 
    digitalWrite(pines[i], LOW);
  }
}
