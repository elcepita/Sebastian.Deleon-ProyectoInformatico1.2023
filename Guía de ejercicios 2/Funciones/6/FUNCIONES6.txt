void activarPines(int pines[], int totalPines) {
  for (int i = 0; i  totalPines; i++) {
    digitalWrite(pines[i], HIGH);
  }
}

void setup() {
  int pines[] = {2, 3, 4, 5, 6};
  int totalPines = sizeof(pines)  sizeof(pines[0]); 
  
  for (int i = 0; i  totalPines; i++) {
    pinMode(pines[i], OUTPUT);
  }
  
  activarPines(pines, totalPines); 
}

}
