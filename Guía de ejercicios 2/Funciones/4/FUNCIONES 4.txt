const int sensorPin = 2; 
const int ledPin = 13; 

void setup() {
  pinMode(sensorPin, INPUT);
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int movimiento = digitalRead(sensorPin);
  
  if (movimiento == HIGH) {
    digitalWrite(ledPin, HIGH); 
    Serial.println("Â¡Movimiento detectado!");
  } else {
    digitalWrite(ledPin, LOW); // Apagar el LED
  }
  
  delay(100); 
}
